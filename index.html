<!DOCTYPE html>

<html>

<head>

  <meta charset="UTF-8">

  <title>Redirecting...</title>

  <script>

    window.onload = function () {

      // Get the referCode from the URL

      const urlParams = new URLSearchParams(window.location.search);

      const referCode = urlParams.get("referCode");


      // Your custom scheme and Play Store URL

      const deepLink = `yourapp://refer?referCode=${referCode}`;

      const playStoreLink = "https://play.google.com/store/apps/details?id=com.yourapp.id";


      // Try opening app

      window.location = deepLink;


      // After delay, fallback to Play Store

      setTimeout(function () {

        window.location = playStoreLink;

      }, 1500);

    };

  </script>

</head>

<body>

  <p>Redirecting you to the app...</p>
